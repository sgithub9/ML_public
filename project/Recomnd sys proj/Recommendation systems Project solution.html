def recommendProductsToUser(userId, df_orig, df_popular_products):
    df_pdts_of_user = df_orig.loc[df_orig['userId']==userId,['productId','rating']].reset_index()
    keys = list(df_pdts_of_user.columns.values)
    i1 = popular_products.set_index(keys).index
    i2 = df_pdts_of_user.set_index(keys).index
    return df_popular_products[~i1.isin(i2)].sort_values(by='rating_count', ascending=False).head()
