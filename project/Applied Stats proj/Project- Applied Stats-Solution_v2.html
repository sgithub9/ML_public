#Checking the presence of outliers in charges column
from scipy import stats

outliers=[]
def detect_outlier(data_1):    
    threshold=3
    mean_1 = np.mean(data_1)
    std_1 =np.std(data_1) 
    
    for y in data_1:
        z_score= (y - mean_1)/std_1 
        if np.abs(z_score) > threshold:
            outliers.append(y)
    return outliers

charges_outliers = detect_outlier(dataframe.values[:,6])
print(charges_outliers)

def draw_distribution_catg_cols(column_name): 
    children_count = dataframe[column_name].value_counts()
    sns.set(style="darkgrid")
    sns.barplot(children_count.index, children_count.values, alpha=0.9)
    plt.title('Frequency Distribution of '+ column_name)
    plt.ylabel('Number of Occurrences', fontsize=12)
    plt.xlabel(column_name, fontsize=12)
    plt.show()
